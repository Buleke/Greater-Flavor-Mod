country_event = {

    id = 888880600

    title = "The 13 State Republic" # New State
    desc = "The states of the replublic now number 13, Our flag must represent this accordingly."
    picture = "texas_state"

    trigger = {
		OR = { tag = USA tag = US2 }
		has_country_flag = Government_Democracy
		NOT = { check_variable = { which = USA_Stars_Count value = 13 }  }
		number_of_states = 1
		NOT = { number_of_states = 15 }
		NOT = { has_global_flag = USAOverride }
		NOT = { has_country_flag = ghost_country }
		war = no
    }

    mean_time_to_happen = {
        days = 1
        modifier = {
            factor = 1
            OR = { USA = { exists = yes } US2 = { exists = yes } }
        }
    }

    option = {
        name = "*Stary Eyed*"
		USA = { set_variable = { which = USA_Stars_Count value = 13 } }
		US2 = { set_variable = { which = USA_Stars_Count value = 13 } }
		random_owned = { limit = { owner = { tag = US2 } } owner = { change_tag = USA } }
		US2 = { all_core = { add_core = USA remove_core = US2 } }
		USA = { government = democracy_5th }
        ai_chance = {
            factor = 100
        }
    }
}
